# 54-二叉搜索树的第k大节点

**题目描述**

给定一颗二叉搜索树，请找出其中的第k大的结点。例如，

``` 
    5 
   / \ 
   3 7 
  /\ /\ 
 2 4 6 8 
```

中，按结点数值大小顺序第三个结点的值为4。

**思路分析**

由于二叉搜索树按照中序遍历后，即是一个递增序列，那么我们用递归法在中序遍历时使用一个count计数，如果count等于k，则直接返回，表示找到第k个大的数。

**我的实现**

```c
/*
struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
    TreeNode(int x) :
            val(x), left(NULL), right(NULL) {
    }
};
*/
class Solution {
public:
    TreeNode* kthNodeCore( TreeNode *pRoot, int k, int& count )
    {
        if( !pRoot )
            return nullptr;
        
        TreeNode* result = nullptr;
        
        if( pRoot->left )
        {
            TreeNode * temp = kthNodeCore( pRoot->left, k, count);;
            if( temp )
                result = temp;
        }         
        
        count++;
        if( count == k )
            result = pRoot;
        
        if( pRoot->right )
        {
            TreeNode * temp = kthNodeCore( pRoot->right, k, count);;
            if( temp )
                result = temp;
        }
        
        //表示已经完全遍历完了，但是还没有count==k，说明k大于count
        return result;
    }
    
    TreeNode* KthNode(TreeNode* pRoot, int k)
    {
        if( !pRoot || k < 1 )
            return nullptr;
        
        int count = 0;
        TreeNode* resultNode = kthNodeCore( pRoot, k, count );
        
        return resultNode;
    }

    
};
```
**二刷**

注意k一定要是引用！

```c++
/*
struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
    TreeNode(int x) :
            val(x), left(NULL), right(NULL) {
    }
};
*/
class Solution {
public:
    TreeNode* KthNode(TreeNode* pRoot, int k)
    {
        return KthNodeCore(pRoot,k);
    }
private:
     TreeNode* KthNodeCore(TreeNode* pRoot, int& k)
    {
        if(!pRoot) return nullptr;
        TreeNode* res = KthNodeCore(pRoot->left,k);
        if(res) return res;
        --k;
        if(k==0) return pRoot;
        return KthNodeCore(pRoot->right,k);
    }   
};
```
